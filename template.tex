% caption 是標題，裡面是虛擬碼。
\begin{algorithm}[caption={Binary Search}, label={alg1}]
function Binary_Search(x, array[])
    l = 0
    r = n - 1 (n是array的大小)
    while l < r
        mid = (l + r) / 2
        if x > array[mid]
            l = mid + 1
        else
            r = mid
    return l
end function
\end{algorithm}
